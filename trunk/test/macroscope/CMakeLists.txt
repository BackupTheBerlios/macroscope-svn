ADD_EXECUTABLE (macroscope
  macroscope.cpp
  htmlrep.cpp
  version.c
)

TARGET_LINK_LIBRARIES (macroscope
  adicpp
  fbcpp
  ksys
  mycpp
  utf8
  ${LIBPDBUTILS}
)

SET_SOURCE_FILES_PROPERTIES (version.c PROPERTIES GENERATED YES)

ADD_DEPENDENCIES (macroscope version macroscope_version)

ADD_CUSTOM_TARGET(macroscope_version
  COMMAND ${VERSION_EXECUTABLE} version.in version.c
  DEPENDS version.in
  WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
)

SET (MACROSCOPE_EXECUTABLE
  "${CMAKE_CURRENT_BINARY_DIR}/${EXECUTABLE_OUTPUT_PATH}${CMAKE_BUILD_TYPE}/macroscope${CMAKE_EXECUTABLE_SUFFIX}"
  CACHE INTERNAL ""
)

FIND_PROGRAM (UPX upx)

IF (PRIVATE_RELEASE AND CMAKE_BUILD_TYPE STREQUAL "Release" AND NOT CMAKE_GENERATOR MATCHES "Win64")
  IF (UPX)
    ADD_CUSTOM_COMMAND (
      TARGET macroscope POST_BUILD
      COMMAND ${UPX} ARGS --brute ${MACROSCOPE_EXECUTABLE}
      WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
      COMMENT ""
    )
  ENDIF (UPX)
ENDIF (PRIVATE_RELEASE AND CMAKE_BUILD_TYPE STREQUAL "Release" AND NOT CMAKE_GENERATOR MATCHES "Win64")

  SET (INSTALL_ETC_DIR "/etc")
